## Installation

```
# npm
npm i @bold-ui/react

# yarn
yarn add @bold-ui/react

# pnpm
pnpm add @bold-ui/react
```

## Usage

```javascript
import { Button } from "@bold-ui/react";
import "@bold-ui/react/dist/css/Button.css"; // Imports styles for Button component only

const MyComponent = () => {
  return (
    <div>
      <Button kind="fill">Bold UI</Button>
    </div>
  );
};
```
